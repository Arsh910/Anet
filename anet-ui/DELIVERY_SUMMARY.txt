BASIC SETUP:

ADD TASK:
  const { addTask } = useKanbanStore()
  addTask('col-todo', {
    title: 'New Task',
    description: 'Task details',
    priority: 'high',
    dueDate: '2026-07-15'
  })

FILTER BY PRIORITY:
  const { setPriorityFilter, getFilteredTasks } = useKanbanStore()
  setPriorityFilter('high')
  const highPriorityTasks = getFilteredTasks()

SEARCH:
  const { setSearchQuery, getFilteredTasks } = useKanbanStore()
  setSearchQuery('landing')
  const results = getFilteredTasks()

MOVE TASK:
  const { moveTask } = useKanbanStore()
  moveTask('task-1', 'col-todo', 'col-in-progress', 0)

EXPORT/IMPORT:
  const { exportState, importState } = useKanbanStore()
  const backup = exportState()
  importState(backup)

================================================================================
ARCHITECTURE
================================================================================

STORE STRUCTURE:
  useKanbanStore (Zustand)
    State:
      - tasks: Map of all tasks
      - columns: Map of all columns
      - columnOrder: Array of column IDs
      - priorityFilter: Current priority filter
      - searchQuery: Current search query

    Actions:
      - Task: add, update, delete, move, reorder
      - Column: add, update, delete, reorder
      - Filter: setPriority, setSearch, getFiltered
      - Utility: reset, clear, export, import

    Persistence:
      - localStorage.setItem('kanban-board-state', state)
      - Auto on init, auto on change
      - Error handling & logging

DATA MODEL:
  Task: {
    id: string,
    title: string,
    description: string,
    priority: 'critical'|'high'|'medium'|'low',
    dueDate: string (ISO),
    createdAt: string (ISO timestamp)
  }

  Column: {
    id: string,
    title: string,
    taskIds: string[]
  }

PERSISTENCE:
  localStorage key: 'kanban-board-state'
  Schema: {
    tasks: { taskId: { ... } },
    columns: { columnId: { ... } },
    columnOrder: [columnId, ...]
  }
